### Geocodificação com Cache e Rate Limiting

```python
import requests
import time

_geo_cache: dict[str, tuple[float, float] | None] = {}
_last_request_time: float = 0.0

def geocode_bairro(bairro: str, cidade: str = "Brasil") -> tuple[float, float] | None:
    """Geocodifica um bairro usando Nominatim. Retorna (lat, lon) ou None."""
    global _last_request_time

    if not bairro:
        return None

    cache_key = f"{bairro},{cidade}".lower()
    if cache_key in _geo_cache:
        return _geo_cache[cache_key]

    # Rate limiting: mínimo 1s entre requisições
    elapsed = time.time() - _last_request_time
    if elapsed < 1.0:
        time.sleep(1.0 - elapsed)

    try:
        resp = requests.get(
            "https://nominatim.openstreetmap.org/search",
            params={"q": f"{bairro}, {cidade}", "format": "json", "limit": 1},
            headers={"User-Agent": "imob-scanner/1.0"},
            timeout=5,
        )
        _last_request_time = time.time()
        data = resp.json()
        if data:
            result = (float(data[0]["lat"]), float(data[0]["lon"]))
            _geo_cache[cache_key] = result
            return result
    except Exception:
        pass

    _geo_cache[cache_key] = None
    return None
```

### Mapeamento de Cores por Portal

```python
PORTAL_COLORS = {
    "ZAP":      [0, 100, 255, 180],
    "VIVAREAL": [0, 200, 100, 180],
    "OLX":      [255, 140, 0, 180],
}

def get_color(portal: str) -> list[int]:
    return PORTAL_COLORS.get(portal.upper(), [180, 180, 180, 180])
```

### Preparação do DataFrame para pydeck

```python
import pydeck as pdk

def build_map_data(df: pd.DataFrame) -> pd.DataFrame:
    """Adiciona colunas lat, lon, color ao DataFrame filtrado."""
    rows = []
    for _, row in df.iterrows():
        bairro = row.get("Bairro") or ""
        if not bairro:
            continue
        coords = geocode_bairro(bairro)
        if coords is None:
            continue
        lat, lon = coords
        rows.append({
            "lat": lat,
            "lon": lon,
            "title": row.get("Título", ""),
            "price": row.get("Preço (R$)", 0),
            "price_fmt": f"R$ {row.get('Preço (R$)', 0):,.0f}",
            "bairro": bairro,
            "portal": row.get("Portal", ""),
            "link": row.get("Link", ""),
            "color": get_color(row.get("Portal", "")),
            "radius": max(50, min(int(row.get("Preço (R$)", 0) / 5000), 500)),
        })
    return pd.DataFrame(rows)
```

### Renderizacao do Mapa com pydeck

```python
def render_map(filtered: pd.DataFrame) -> None:
    map_df = build_map_data(filtered)

    if map_df.empty:
        st.info("Nenhum imóvel com coordenadas disponível para os filtros selecionados.")
        return

    layer = pdk.Layer(
        "ScatterplotLayer",
        data=map_df,
        get_position=["lon", "lat"],
        get_color="color",
        get_radius="radius",
        pickable=True,
        opacity=0.8,
        stroked=True,
        filled=True,
        radius_min_pixels=6,
        radius_max_pixels=40,
    )

    view_state = pdk.ViewState(
        latitude=map_df["lat"].mean(),
        longitude=map_df["lon"].mean(),
        zoom=11,
        pitch=0,
    )

    tooltip = {
        "html": (
            "<b>{title}</b><br/>"
            "Preco: {price_fmt}<br/>"
            "Bairro: {bairro}<br/>"
            "Portal: {portal}<br/>"
            "<a href='{link}' target='_blank'>Ver anuncio</a>"
        ),
        "style": {"backgroundColor": "white", "color": "black", "padding": "8px"},
    }

    st.pydeck_chart(pdk.Deck(layers=[layer], initial_view_state=view_state, tooltip=tooltip))
    st.caption(f"{len(map_df)} imóveis exibidos no mapa.")
```
